📊 เชื่อม Google Sheet

ขั้นตอน:
1. เปิด Google Sheet ใหม่ของคุณ
2. เมนู Extensions → Apps Script
3. ลบโค้ดเดิม → วางโค้ดด้านล่าง → Save
4. Deploy → New deployment → Web app
5. Execute as: Me · Access: Anyone
6. Copy URL มาใส่ด้านล่าง
📋 แตะเพื่อคัดลอก Apps Script:
function doPost(e){ try{ var d=JSON.parse(e.postData.contents); var ss=SpreadsheetApp.getActiveSpreadsheet(); var sh=ss.getSheetByName('DonNote')||ss.insertSheet('DonNote'); if(sh.getLastRow()===0) sh.appendRow(['วันที่','ประเภท','หมวด','ไอคอน','จำนวน','หมายเหตุ']); if(d.action==='sync'){ if(sh.getLastRow()>1) sh.deleteRows(2,sh.getLastRow()-1); d.entries.forEach(function(r){ sh.appendRow([r.date,r.type==='expense'?'รายจ่าย':'รายรับ',r.catName,r.catIcon,r.amount,r.note]); }); } return ContentService.createTextOutput(JSON.stringify({status:'ok'})).setMimeType(ContentService.MimeType.JSON); }catch(err){ return ContentService.createTextOutput(JSON.stringify({status:'error',msg:err.toString()})).setMimeType(ContentService.MimeType.JSON); } } function doGet(){return ContentService.createTextOutput(JSON.stringify({status:'ok'})).setMimeType(ContentService.MimeType.JSON);}
